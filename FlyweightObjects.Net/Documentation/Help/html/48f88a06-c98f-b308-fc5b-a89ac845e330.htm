CacheManager Class
FlyweightObjects.NET
Provides the default caching functionality to be used by the DataContext. Note that the domain model must be adorned with a DataTableAttribute and the attribute itself must have EnableCaching set to true.

Examples

CopyC#
[DataTable("Production.Product", EnableCaching=true, CacheTimeout=300)]
public class MyProduct : IRedundancyCheck
{
    [DataColumn("ProductID", Identifier = true)]
    public int ProductID { get; set; }

    [DataColumn("ModifiedDate")]
    public DateTime ModifiedDate { get; set; }

    [PropertyTorage]
    public string Checksum { get; set; }

    [Computed]
    public bool IsChanged
    {
        get { return new ChecksumBuilder().BuildChecksum(this) != this.Checksum; }
    }
}

class Program
{
    static void Main(string[] args)
    {
        using (DataContext context = new DataContext(@"Integrated Security=SSPI;Initial Catalog=AdventureWorks;Data Source=localhost\SQL2K5;", StorageProviderType.SqlServer))
        {
            context.CacheManager.Enabled = true;
            MyProduct p1 = context.Select<MyProduct>(1).ToSingle();
            Console.Writeline("Elapsed Time: {0}", context.ElapsedMilliseconds);
            MyProduct p2 = context.Select<MyProduct>(1).ToSingle();
            Console.Writeline("Elapsed Time: {0}", context.ElapsedMilliseconds);
        }     
    }
}
