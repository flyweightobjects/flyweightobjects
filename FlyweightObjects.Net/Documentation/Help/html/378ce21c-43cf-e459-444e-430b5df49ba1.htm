Examples

The following example uses a Product class from the Microsoft AdventureWorks SQL Server 2005 sample database.
CopyC#
using (DataContext context = new DataContext(@"Integrated Security=SSPI;Initial Catalog=AdventureWorks;Data Source=localhost\SQL2K5;", StorageProviderType.SqlServer))
{
    // Retrieve the top 100 products
    var products = context.Select<Product>(100);
    Console.WriteLine("Count is {0}", products.Count<Product>());

    // Change the ModifiedDate of each product
    foreach (product product in products)
    {
        products.ModifiedDate = DateTime.Now;
    }

    // Update the products
    context.Update<Product>(products);
